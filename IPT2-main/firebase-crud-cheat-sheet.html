Get Documents with Pagination

                

                  
// Paginated query
import { startAfter, startAt } from 'firebase/firestore';

const getUsersPaginated = async (pageSize = 10, lastDoc = null) => {
  try {
    let q = query(
      collection(db, "users"),
      orderBy("createdAt", "desc"),
      limit(pageSize)
    );
    
    if (lastDoc) {
      q = query(q, startAfter(lastDoc));
    }
    
    const querySnapshot = await getDocs(q);
    const users = [];
    let lastDocument = null;
    
    querySnapshot.forEach((doc) => {
      users.push({ id: doc.id, ...doc.data() });
      lastDocument = doc;
    });
    
    return { users, lastDocument };
  } catch (error) {
    console.error("Error getting paginated documents: ", error);
    return { users: [], lastDocument: null };
  }
};

Delete Documents Matching Query

                

                  
// Delete all documents matching a query
const deleteInactiveUsers = async () => {
  try {
    // First, get documents to delete
    const q = query(
      collection(db, "users"),
      where("lastLogin", "<", new Date(Date.now() - 90 * 24 * 60 * 60 * 1000)) // 90 days ago
    );
    
    const querySnapshot = await getDocs(q);
    
    // Delete in batches (Firestore batch limit: 500 operations)
    const batch = writeBatch(db);
    let batchCount = 0;
    
    for (const document of querySnapshot.docs) {
      batch.delete(document.ref);
      batchCount++;
      
      // Commit batch when it reaches 500 operations
      if (batchCount === 500) {
        await batch.commit();
        batchCount = 0;
      }
    }
    
    // Commit remaining operations
    if (batchCount > 0) {
      await batch.commit();
    }
    
    console.log(`${querySnapshot.docs.length} inactive users deleted`);
    return querySnapshot.docs.length;
  } catch (error) {
    console.error("Error deleting inactive users: ", error);
    return 0;
  }
};

Delete with Confirmation

                

                  
// Delete with user confirmation and error handling
const deleteWithConfirmation = async (userIds, confirmationText) => {
  if (confirmationText !== "DELETE") {
    throw new Error("Confirmation text must be 'DELETE'");
  }
  
  try {
    // Get user data before deletion for logging
    const usersToDelete = [];
    for (const userId of userIds) {
      const userDoc = await getDoc(doc(db, "users", userId));
      if (userDoc.exists()) {
        usersToDelete.push({ id: userId, ...userDoc.data() });
      }
    }
    
    // Create deletion batch
    const batch = writeBatch(db);
    userIds.forEach(userId => {
      batch.delete(doc(db, "users", userId));
    });
    
    // Log deletion attempt
    await addDoc(collection(db, "deletionLogs"), {
      deletedUsers: usersToDelete,
      deletedAt: new Date(),
      deletedBy: "admin" // or current user ID
    });
    
    // Commit deletion
    await batch.commit();
    
    console.log(`${userIds.length} users deleted and logged`);
    return { success: true, deletedCount: userIds.length };
  } catch (error) {
    console.error("Error in confirmed deletion: ", error);
    return { success: false, error: error.message };
  }
};

Document Listener

              

                
// Listen to real-time changes on a single document
const listenToUser = (userId, callback) => {
  const unsubscribe = onSnapshot(
    doc(db, "users", userId),
    (doc) => {
      if (doc.exists()) {
        callback({ id: doc.id, ...doc.data() });
      } else {
        callback(null);
      }
    },
    (error) => {
      console.error("Error listening to user: ", error);
    }
  );
  
  // Return unsubscribe function
  return unsubscribe;
};

Large Batch Processing

              

                
// Handle large datasets with batch size limits
const processLargeDataset = async (data) => {
  const BATCH_SIZE = 500; // Firestore limit
  
  for (let i = 0; i < data.length; i += BATCH_SIZE) {
    const batch = writeBatch(db);
    const batchData = data.slice(i, i + BATCH_SIZE);
    
    batchData.forEach((item, index) => {
      const docRef = doc(collection(db, "users"));
      batch.set(docRef, {
        ...item,
        batchNumber: Math.floor(i / BATCH_SIZE) + 1,
        indexInBatch: index,
        createdAt: new Date()
      });
    });
    
    try {
      await batch.commit();
      console.log(`Batch ${Math.floor(i / BATCH_SIZE) + 1} completed`);
    } catch (error) {
      console.error(`Batch ${Math.floor(i / BATCH_SIZE) + 1} failed:`, error);
    }
  }
};

              

            

          

        


        
        

          
Transactions


          

            

              
Basic Transaction

              

                
// Atomic transaction example
import { runTransaction } from 'firebase/firestore';

const transferPoints = async (fromUserId, toUserId, points) => {
  try {
    await runTransaction(db, async (transaction) => {
      // Read both users
      const fromUserRef = doc(db, "users", fromUserId);
      const toUserRef = doc(db, "users", toUserId);
      
      const fromUserDoc = await transaction.get(fromUserRef);
      const toUserDoc = await transaction.get(toUserRef);
      
      if (!fromUserDoc.exists()) {
        throw new Error("Source user does not exist!");
      }
      
      if (!toUserDoc.exists()) {
        throw new Error("Destination user does not exist!");
      }
      
      const fromUserPoints = fromUserDoc.data().points || 0;
      
      if (fromUserPoints < points) {
        throw new Error("Insufficient points!");
      }
      
      // Update both users
      transaction.update(fromUserRef, {
        points: fromUserPoints - points
      });
      
      transaction.update(toUserRef, {
        points: (toUserDoc.data().points || 0) + points
      });
    });
    
    console.log("Points transferred successfully!");
  } catch (error) {
    console.error("Transaction failed: ", error);
  }
};

      for (const item of items) {
        const productRef = doc(db, "products", item.productId);
        const productDoc = await transaction.get(productRef);
        
        if (!productDoc.exists()) {
          throw new Error(`Product ${item.productId} not found`);
        }
        
        const stock = productDoc.data().stock || 0;
        if (stock < item.quantity) {
          throw new Error(`Insufficient stock for ${item.productId}`);
        }
        
        results.push({
          productRef,
          currentStock: stock,
          orderedQuantity: item.quantity
        });
      }
      
      // Create order
      const orderRef = doc(collection(db, "orders"));
      transaction.set(orderRef, {
        userId,
        items,
        status: "pending",
        total: items.reduce((sum, item) => sum + (item.price * item.quantity), 0),
        createdAt: new Date()
      });
      
      // Update inventory
      results.forEach(({ productRef, currentStock, orderedQuantity }) => {
        transaction.update(productRef, {
          stock: currentStock - orderedQuantity,
          lastOrdered: new Date()
        });
      });
      
      return orderRef.id;
    });
    
    console.log("Order created with ID: ", result);
    return result;
  } catch (error) {
    console.error("Order creation failed: ", error);
    return null;
  }
};

// Comprehensive error handling and validation
class UserService {
  constructor(db) {
    this.db = db;
    this.collection = collection(db, "users");
  }
  
  // Validate user data
  validateUser(userData) {
    const errors = [];
    
    if (!userData.email || !userData.email.includes('@')) {
      errors.push('Valid email is required');
    }
    
    if (!userData.name || userData.name.trim().length < 2) {
      errors.push('Name must be at least 2 characters');
    }
    
    if (userData.age && (userData.age < 0 || userData.age > 150)) {
      errors.push('Age must be between 0 and 150');
    }
    
    return errors;
  }
  
  // Create user with validation
  async createUser(userData) {
    const errors = this.validateUser(userData);
    if (errors.length > 0) {
      throw new Error(`Validation failed: ${errors.join(', ')}`);
    }
    
    try {
      // Check if email already exists
      const existingUser = await this.getUserByEmail(userData.email);
      if (existingUser) {
        throw new Error('User with this email already exists');
      }
      
      const docRef = await addDoc(this.collection, {
        ...userData,
        createdAt: new Date(),
        updatedAt: new Date()
      });
      
      return { success: true, id: docRef.id };
    } catch (error) {
      console.error('Error creating user:', error);
      return { success: false, error: error.message };
    }
  }
  
  // Get user by email with caching
  async getUserByEmail(email) {
    try {
      const q = query(this.collection, where("email", "==", email), limit(1));
      const querySnapshot = await getDocs(q);
      
      if (!querySnapshot.empty) {
        const doc = querySnapshot.docs[0];
        return { id: doc.id, ...doc.data() };
      }
      
      return null;
    } catch (error) {
      console.error('Error getting user by email:', error);
      throw error;
    }
  }
}

// Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can only read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null 
        && request.auth.uid == userId
        && validateUserData(resource.data);
    }
    
    // Public read, authenticated write for posts
    match /posts/{postId} {
      allow read: if true;
      allow create, update: if request.auth != null 
        && request.auth.uid == resource.data.authorId;
      allow delete: if request.auth != null 
        && (request.auth.uid == resource.data.authorId 
            || request.auth.token.admin == true);
    }
    
    // Private subcollections
    match /users/{userId}/private/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
  
  // Validation function
  function validateUserData(data) {
    return data.keys().hasAll(['name', 'email', 'createdAt']) &&
           data.name is string &&
           data.name.size() >= 2 &&
           data.email.matches('.*@.*\\..*');
  }
}
